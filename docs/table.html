    <table class="table"></table>
</div>

表头固定

表格列的宽度由表格宽度决定，不由内容决定，并且需要设置tbody的高度，需要动态计算滚动条的宽度防止错位。

示例代码：
/** 动态计算滚动条的宽度 */
function getScrollbarWidth() {
    var scroller = document.createElement("div"), body = document.body, scrollWidth = 0;

    scroller.style.cssText = "width: 100px; height: 100px; overflow: scroll; visibility: hidden;";
    body.appendChild(scroller);
    scrollWidth = scroller.offsetWidth - scroller.clientWidth;
    body.removeChild(scroller);

    return scrollWidth
}

<div class="table-container">
    <div class="table-header">

            <colgroup>
                <col style="width: 200px;">
                ...
            </colgroup>
            <thead></thead>
        </table>
    </div>
    <div class="table-body" style="max-height: 248px;">
        <table class="table">
            <colgroup>
                <col style="width: 200px;">
                ...
            </colgroup>
            <tbody></tbody>
        </table>
    </div>
</div>
